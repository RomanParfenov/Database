

CREATE TABLE IF NOT EXISTS genre (genreid serial primary key, genrename varchar(100) not null unique);

CREATE TABLE IF NOT EXISTS artist (artistid serial primary key, name varchar(100) not null unique, genreid 
integer references genre(genreid));

CREATE TABLE IF NOT EXISTS albums (albumid serial primary key, artistid integer references artist(artistid),
album_name varchar(100) not null unique, year date);

CREATE TABLE IF NOT EXISTS track (trackid serial primary key, albumid integer references albums(albumid), 
title varchar(100) not null unique, time text);
